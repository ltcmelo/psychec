37 /*
38  * There are 2 coarse variations of AST nodes:
39  *
40  * 1 - those that inherit directly from `SyntaxNode', creating a syntactic
41  * category (6.1-1) for the syntax in question;
42  * 2 - those that belong to an specific category.
43  *
44  * Macros prefixed with `AST_G_NODE' apply on 1, while macros prefixed
45  * with `AST_NODE' apply on 2.
46  */
47 #define AST_G_NODE__COMMON__(NODE) \
48  AST__COMMON__(NODE,) \
49  using BaseSyntax = SyntaxNode;
50 
51 #define AST_NODE__COMMON__(NODE, BASE_NODE) \
52  AST__COMMON__(NODE, BASE_NODE) \
53  using BaseSyntax = BASE_NODE##Syntax;
54 
55 /*
56  * The class of an AST node further varies in respect to whether:
57  *
58  * - it serves exclusively as base type (not visited during AST traversal);
59  * - it is a type for which a single concrete syntax kind exists;
60  * - it is a type for which multiple concrete syntax kinds exist.
61  *
62  * In the last case, the construct for the AST node in question requires
63  * an specific `SyntaxKind' argument.
64  */
65 #define AST_G_NODE(NODE) \
66  AST_G_NODE__COMMON__(NODE) \
67  using SyntaxNode::SyntaxNode;
68 #define AST_G_NODE_1K(NODE) \
69  AST_G_NODE__COMMON__(NODE) \
70  NODE##Syntax(SyntaxTree* tree) : SyntaxNode(tree, NODE) {} \
71  DISPATCH_VISIT(NODE)
72 #define AST_G_NODE_NK(NODE) \
73  AST_G_NODE__COMMON__(NODE) \
74  NODE##Syntax(SyntaxTree* tree, SyntaxKind kind) : SyntaxNode(tree, kind) {} \
75  DISPATCH_VISIT(NODE)
76 
77 #define AST_NODE(NODE, BASE_NODE) \
78  AST_NODE__COMMON__(NODE, BASE_NODE) \
79  using BASE_NODE##Syntax::BASE_NODE##Syntax;
80 #define AST_NODE_1K(NODE, BASE_NODE) \
81  AST_NODE__COMMON__(NODE, BASE_NODE) \
82  NODE##Syntax(SyntaxTree* tree) : BASE_NODE##Syntax(tree, NODE) {} \
83  DISPATCH_VISIT(NODE)
84 #define AST_NODE_NK(NODE, BASE_NODE) \
85  AST_NODE__COMMON__(NODE, BASE_NODE) \
86  NODE##Syntax(SyntaxTree* tree, SyntaxKind kind) : BASE_NODE##Syntax(tree, kind) {} \
87  DISPATCH_VISIT(NODE)

138 /*
139  * The default implementation of the visitor dispatching function for
140  * vising the `this' node.
141  */
142 #define DISPATCH_VISIT(NODE) \
143  protected: \
144  virtual SyntaxVisitor::Action dispatchVisit(SyntaxVisitor* visitor) const override \
145  { return visitor->visit##NODE(this); }
146 
147 /*
148  * The default implementation of the function that gather the child
149  * nodes and tokens of the `this' node.
150  */
151 #define CHILD_NODES_AND_TOKENS(CHILDREN_SYNTAX) \
152  public: \
153  virtual std::vector<SyntaxHolder> childNodesAndTokens() const override \
154  { auto self = { CHILDREN_SYNTAX }; \
155  return merge(BaseSyntax::childNodesAndTokens(), self); }

160 inline std::vector<SyntaxHolder> merge(std::vector<SyntaxHolder>&& a,
161  std::vector<SyntaxHolder>&& b)
162 {
163  a.insert(a.end(), b.begin(), b.end());
164  return std::move(a);
165 }
