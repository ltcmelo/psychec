38 /*
39  * There are 2 coarse variations of AST nodes:
40  *
41  * 1 - those that inherit directly from `SyntaxNode', creating a syntactic
42  * category (6.1-1) for the syntax in question;
43  * 2 - those that belong to an specific category.
44  *
45  * Macros prefixed with `AST_G_NODE' apply on 1, while macros prefixed
46  * with `AST_NODE' apply on 2.
47  */
48 #define AST_G_NODE__COMMON__(NODE) \
49  AST__COMMON__(NODE,) \
50  using BaseSyntax = SyntaxNode;
51 
52 #define AST_NODE__COMMON__(NODE, BASE_NODE) \
53  AST__COMMON__(NODE, BASE_NODE) \
54  using BaseSyntax = BASE_NODE##Syntax;
55 
56 /*
57  * The class of an AST node further varies in respect to whether:
58  *
59  * - it serves exclusively as base type (not visited during AST traversal);
60  * - it is a type for which a single concrete syntax kind exists;
61  * - it is a type for which multiple concrete syntax kinds exist.
62  *
63  * In the last case, the construct for the AST node in question requires
64  * an specific `SyntaxKind' argument.
65  */
66 #define AST_G_NODE(NODE) \
67  AST_G_NODE__COMMON__(NODE) \
68  using SyntaxNode::SyntaxNode;
69 #define AST_G_NODE_1K(NODE) \
70  AST_G_NODE__COMMON__(NODE) \
71  NODE##Syntax(SyntaxTree* tree) : SyntaxNode(tree, SyntaxKind::NODE) {} \
72  DISPATCH_VISIT(NODE)
73 #define AST_G_NODE_NK(NODE) \
74  AST_G_NODE__COMMON__(NODE) \
75  NODE##Syntax(SyntaxTree* tree, SyntaxKind kind) : SyntaxNode(tree, kind) {} \
76  DISPATCH_VISIT(NODE)
77 
78 #define AST_NODE(NODE, BASE_NODE) \
79  AST_NODE__COMMON__(NODE, BASE_NODE) \
80  using BASE_NODE##Syntax::BASE_NODE##Syntax;
81 #define AST_NODE_1K(NODE, BASE_NODE) \
82  AST_NODE__COMMON__(NODE, BASE_NODE) \
83  NODE##Syntax(SyntaxTree* tree) : BASE_NODE##Syntax(tree, SyntaxKind::NODE) {} \
84  DISPATCH_VISIT(NODE)
85 #define AST_NODE_NK(NODE, BASE_NODE) \
86  AST_NODE__COMMON__(NODE, BASE_NODE) \
87  NODE##Syntax(SyntaxTree* tree, SyntaxKind kind) : BASE_NODE##Syntax(tree, kind) {} \
88  DISPATCH_VISIT(NODE)

139 /*
140  * The default implementation of the visitor dispatching function for
141  * vising the `this' node.
142  */
143 #define DISPATCH_VISIT(NODE) \
144  protected: \
145  virtual SyntaxVisitor::Action dispatchVisit(SyntaxVisitor* visitor) const override \
146  { return visitor->visit##NODE(this); }

148 /*
149  * The default implementation of the function that gather the child
150  * nodes and tokens of the `this' node.
151  */
152 #define CHILD_NODES_AND_TOKENS(CHILDREN_SYNTAX) \
153  public: \
154  virtual std::vector<SyntaxHolder> childNodesAndTokens() const override \
155  { auto self = { CHILDREN_SYNTAX }; \
156  return merge(BaseSyntax::childNodesAndTokens(), self); }

161 inline std::vector<SyntaxHolder> merge(std::vector<SyntaxHolder>&& a,
162  std::vector<SyntaxHolder>&& b)
163 {
164  a.insert(a.end(), b.begin(), b.end());
165  return std::move(a);
166 }
