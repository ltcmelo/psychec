212 template <class SymT, class... SymTArgs>
213 SymT* Binder::bindAndPushSymbol(const SyntaxNode* node, SymTArgs... args)
214 {
215  std::unique_ptr<SymT> sym(new SymT(tree_,
216  syms_.top(),
217  scopes_.top(),
218  std::forward<SymTArgs>(args)...));
219  auto rawSym = static_cast<SymT*>(semaModel_->keepBinding(node, std::move(sym)));
220  pushSymbol(rawSym);
221  return rawSym;
222 }

227  std::unique_ptr<TyT> ty(new TyT(std::forward<TyTArgs>(args)...));
