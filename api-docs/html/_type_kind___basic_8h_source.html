47 enum class PSY_C_API BasicTypeKind : std::uint8_t
48 {
49  Char, // char
50  Char_S, // signed char
51  Char_U, // unsigned char
52  Short, Short_S, // short, signed short, short int, or signed short int
53  Short_U, // unsigned short, or unsigned short int
54  Int, Int_S, // int, signed, or signed int
55  Int_U, // unsigned, or unsigned int
56  Long, Long_S, // long, signed long, long int, or signed long int
57  Long_U, // unsigned long, or unsigned long int
58  LongLong, LongLong_S, // long long, signed long long, long long int, or signed long long int
59  LongLong_U, // unsigned long long, or unsigned long long int
60  Float,
61  Double,
62  LongDouble,
63  Bool,
64  FloatComplex,
65  DoubleComplex,
66  LongDoubleComplex
67 };

69 inline std::string PSY_C_API to_string(BasicTypeKind basicTyK)
70 {
71  switch (basicTyK) {
72  case BasicTypeKind::Char:
73  return "char";
74  case BasicTypeKind::Char_S:
75  return "signed char";
76  case BasicTypeKind::Char_U:
77  return "unsigned char";
78  case BasicTypeKind::Short:
79  return "short";
80  case BasicTypeKind::Short_S:
81  return "signed short";
82  case BasicTypeKind::Short_U:
83  return "unsigned short";
84  case BasicTypeKind::Int:
85  return "int";
86  case BasicTypeKind::Int_S:
87  return "signed int";
88  case BasicTypeKind::Int_U:
89  return "unsigned int";
90  case BasicTypeKind::Long:
91  return "long";
92  case BasicTypeKind::Long_S:
93  return "signed long";
94  case BasicTypeKind::Long_U:
95  return "unsigned long";
96  case BasicTypeKind::LongLong:
97  return "long long";
98  case BasicTypeKind::LongLong_S:
99  return "signed long long";
100  case BasicTypeKind::LongLong_U:
101  return "unsigned long long";
102  case BasicTypeKind::Float:
103  return "float";
104  case BasicTypeKind::Double:
105  return "double";
106  case BasicTypeKind::LongDouble:
107  return "long double";
108  case BasicTypeKind::Bool:
109  return "_Bool";
110  case BasicTypeKind::FloatComplex:
111  return "float _Complex";
112  case BasicTypeKind::DoubleComplex:
113  return "double _Complex";
114  case BasicTypeKind::LongDoubleComplex:
115  return "long double _Complex";
116  }
117 }
