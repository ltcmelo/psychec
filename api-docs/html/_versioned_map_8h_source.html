121 template <class KeyT, class ValueT>
122 void VersionedMap<KeyT, ValueT>::insertOrAssign_CORE(const KeyT& key,
123  ValueT&& value)
124 {
125  // We don't want to use operator[] because this would require ValueT to
126  // have a default constructor (Range, for instance, doesn't have one, and
127  // this is by design). C++17 has already insert_or_assign.
128  auto it = map_.find(key);
129  if (it != map_.end())
130  it->second = std::move(value);
131  else
132  map_.insert(std::make_pair(key, std::move(value)));
133 }

135 template <class KeyT, class ValueT>
136 void VersionedMap<KeyT, ValueT>::storeCommand(const Command& command)
137 {
138  commands_.push_back(command);
139  reverts_.insert(std::make_pair(++revisionCnt_, curRevision_));
140  curRevision_ = revisionCnt_;
141 }

143 template <class KeyT, class ValueT>
144 void VersionedMap<KeyT, ValueT>::applyRevision(uint32_t revision)
145 {
146  std::vector<uint32_t> ordered;
147  auto it = reverts_.find(revision);
148  while (it != reverts_.end()) {
149  ordered.push_back(it->second);
150  it = reverts_.find(it->second);
151  }
152 
153  curRevision_ = revision;
154  if (ordered.empty())
155  return; // Nothing to do.
156 
157  std::reverse(ordered.begin(), ordered.end());
158  map_.clear();
159  for (auto current : ordered) {
160  const auto& command = commands_[current];
161  ValueT value = command.value_;
162  if (command.opCode_ == Command::Insert)
163  insertOrAssign_CORE(command.key_, std::move(value));
164  }
165 }
