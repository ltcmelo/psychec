131 inline std::ostream& operator<<(std::ostream& os, const Range& range)
132 {
133  const auto& l = range.lower_->evaluate();
134  const auto& u = range.upper_->evaluate();
135  auto show = [&os] (const AbstractValue::Bound& bound, char sign) {
136  if (bound.kind_ == AbstractValue::Bound::Undefined)
137  os << sign << std::numeric_limits<double>::infinity();
138  else
139  os << bound.number_;
140  };
141  os << "[";
142  show(l, '-');
143  os << ", ";
144  show(u, '+');
145  os << "]";
146  return os;
147 }
